from collections import defaultdict

# Read n and m
n, m = map(int, input().split())

# Use a defaultdict of lists to store indices of words in group A
group_a = defaultdict(list)

for i in range(1, n + 1):
    word = input()
    group_a[word].append(str(i))  # Store as string for easy joining later

# Process group B
for _ in range(m):
    word_b = input()
    if word_b in group_a:
        print(" ".join(group_a[word_b]))
    else:
        print("-1")
